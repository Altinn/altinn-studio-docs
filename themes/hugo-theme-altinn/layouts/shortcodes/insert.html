{{/* First param should be full path to the file. Following parameters will be replaces based on a 0-index. Placeholders must use format {0}, {1}... {n} Usage: {{% insert [full file path] [variable 0]... [variable n]" %}} */}} {{ $filePath := .Get 0 }} {{ if fileExists $filePath }} {{ $content := readFile $filePath }} {{ if $content }} {{ $substring := "---" }} {{ $parts := split $content $substring }} {{/* The first "---" delimited block is the front matter (preceded by an empty segment), so skip the first two segments and rejoin the rest with "---". Rejoining preserves any "---" that belongs to the content itself, such as markdown table separator rows (|---|---|). */}} {{ if ge (len $parts) 3 }} {{ $content = delimit (after 2 $parts) $substring }} {{ end }} {{ range $index, $param := .Params }} {{ if ne $index 0 }} {{ $content = replace $content (printf "{%d}" (sub $index 1)) $param }} {{ end }} {{- end -}} {{ $content | markdownify}} {{ end }} {{ else }} {{ errorf "Could not find section content to be inserted in %s. File does not exist: %s" .Page.RelPermalink $filePath }} {{ end }}
